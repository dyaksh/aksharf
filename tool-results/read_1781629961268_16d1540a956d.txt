    72→function ParticleCanvas({ color1 = '#5d2c86', color2 = '#D4AF37' }: { color1?: string; color2?: string }) {
    73→  const canvasRef = useRef<HTMLCanvasElement>(null);
    74→  const animRef = useRef<number>(0);
    75→
    76→  useEffect(() => {
    77→    const canvas = canvasRef.current;
    78→    if (!canvas) return;
    79→    const ctx = canvas.getContext('2d');
    80→    if (!ctx) return;
    81→
    82→    const resize = () => {
    83→      canvas.width = canvas.offsetWidth * 2;
    84→      canvas.height = canvas.offsetHeight * 2;
    85→      ctx.scale(2, 2);
    86→    };
    87→    resize();
    88→    window.addEventListener('resize', resize);
    89→
    90→    interface Particle {
    91→      x: number;
    92→      y: number;
    93→      vx: number;
    94→      vy: number;
    95→      r: number;
    96→      opacity: number;
    97→      color: string;
    98→      pulse: number;
    99→      pulseSpeed: number;
   100→    }
   101→
   102→    const particles: Particle[] = Array.from({ length: 50 }, () => ({
   103→      x: Math.random() * canvas.offsetWidth,
   104→      y: Math.random() * canvas.offsetHeight,
   105→      vx: (Math.random() - 0.5) * 0.3,
   106→      vy: (Math.random() - 0.5) * 0.3,
   107→      r: Math.random() * 2 + 0.5,
   108→      opacity: Math.random() * 0.4 + 0.1,
   109→      color: Math.random() > 0.5 ? color1 : color2,
   110→      pulse: Math.random() * Math.PI * 2,
   111→      pulseSpeed: Math.random() * 0.02 + 0.01,
   112→    }));
   113→
   114→    const animate = () => {
   115→      ctx.clearRect(0, 0, canvas.offsetWidth, canvas.offsetHeight);
   116→      particles.forEach((p) => {
   117→        p.x += p.vx;
   118→        p.y += p.vy;
   119→        p.pulse += p.pulseSpeed;
   120→        if (p.x < 0 || p.x > canvas.offsetWidth) p.vx *= -1;
   121→        if (p.y < 0 || p.y > canvas.offsetHeight) p.vy *= -1;
   122→        const alpha = p.opacity * (0.6 + 0.4 * Math.sin(p.pulse));
   123→        ctx.beginPath();
   124→        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
   125→        ctx.fillStyle = p.color;
   126→        ctx.globalAlpha = alpha;
   127→        ctx.fill();
   128→
   129→        // glow
   130→        ctx.beginPath();
   131→        ctx.arc(p.x, p.y, p.r * 3, 0, Math.PI * 2);
   132→        const grad = ctx.createRadialGradient(p.x, p.y, 0, p.x, p.y, p.r * 3);
   133→        grad.addColorStop(0, p.color);
   134→        grad.addColorStop(1, 'transparent');
   135→        ctx.fillStyle = grad;
   136→        ctx.globalAlpha = alpha * 0.3;
   137→        ctx.fill();
   138→      });
   139→      ctx.globalAlpha = 1;
   140→      animRef.current = requestAnimationFrame(animate);
   141→    };
   142→    animate();
   143→
   144→    return () => {
   145→      cancelAnimationFrame(animRef.current);
   146→      window.removeEventListener('resize', resize);
   147→    };
   148→  }, [color1, color2]);
   149→
   150→  return (
   151→    <canvas
   152→      ref={canvasRef}
   153→      className="absolute inset-0 w-full h-full pointer-events-none"
   154→      style={{ opacity: 0.7 }}
   155→      aria-hidden="true"
   156→    />
   157→  );
   158→}

   527→
   528→export default function ConcentricCircles3D() {
   529→  const containerRef = useRef<HTMLDivElement>(null);
   530→  const sceneRef = useRef<HTMLDivElement>(null);
   531→  const isInView = useInView(containerRef, { once: true, margin: '-80px' });
   532→  const [hoveredIndex, setHoveredIndex] = useState<number | null>(null);
   533→  const [activeIndex, setActiveIndex] = useState<number | null>(null);
   534→  const [mousePos, setMousePos] = useState({ x: 0, y: 0 });
   535→  const [isMobile, setIsMobile] = useState(false);
   536→
   537→  // Detect mobile
   538→  useEffect(() => {
   539→    const check = () => setIsMobile(window.innerWidth < 768);
   540→    check();
   541→    window.addEventListener('resize', check);
   542→    return () => window.removeEventListener('resize', check);
   543→  }, []);
   544→
   545→  // Mouse parallax for desktop
   546→  const handleMouseMove = useCallback(
   547→    (e: React.MouseEvent) => {
   548→      if (isMobile) return;
   549→      const rect = containerRef.current?.getBoundingClientRect();
   550→      if (!rect) return;
   551→      const x = ((e.clientX - rect.left) / rect.width - 0.5) * 2;
   552→      const y = ((e.clientY - rect.top) / rect.height - 0.5) * 2;
   553→      setMousePos({ x, y });
   554→    },
   555→    [isMobile]
   556→  );
   557→
   558→  const handleMouseLeave = useCallback(() => {
   559→    setMousePos({ x: 0, y: 0 });
   560→  }, []);
   561→
   562→  const selectedService = activeIndex !== null ? services[activeIndex] : null;
   563→
   564→  // Responsive orbit radii
   565→  const outerRadius = isMobile ? 155 : 195;
   566→  const innerRadius = isMobile ? 100 : 130;
   567→
   568→  // Pre-compute node positions with rounded values to avoid hydration mismatch
   569→  const outerRadiusPct = isMobile ? 38 : 39;
   570→  const innerRadiusPct = isMobile ? 24 : 26;
   571→  const nodePositions = useMemo(() => {
   572→    return services.map((_, i) => {
   573→      const angle = (i * 60 - 90) * (Math.PI / 180);
   574→      return {
   575→        inner: {
   576→          x: Math.round((50 + innerRadiusPct * Math.cos(angle)) * 1000) / 1000,
   577→          y: Math.round((50 + innerRadiusPct * Math.sin(angle)) * 1000) / 1000,
   578→        },
   579→        outer: {
   580→          x: Math.round((50 + outerRadiusPct * Math.cos(angle)) * 1000) / 1000,
   581→          y: Math.round((50 + outerRadiusPct * Math.sin(angle)) * 1000) / 1000,
   582→        },
   583→      };
   584→    });
   585→  }, [innerRadiusPct, outerRadiusPct]);
   586→
   587→  return (
   588→    <div ref={containerRef} className="relative w-full max-w-4xl mx-auto" onMouseMove={handleMouseMove} onMouseLeave={handleMouseLeave}>
   589→      {/* ── 3D SCENE CONTAINER ── */}
   590→      <div className="relative w-full" style={{ paddingBottom: '100%' }}>
   591→        <div
   592→          ref={sceneRef}
   593→          className="absolute inset-0"
   594→          style={{
   595→            perspective: isMobile ? '1200px' : '900px',
   596→            perspectiveOrigin: '50% 50%',
   597→            transformStyle: 'preserve-3d',
   598→          }}
   599→        >
   600→          {/* 3D tilted scene */}
   601→          <motion.div
   602→            className="absolute inset-0"
   603→            style={{ transformStyle: 'preserve-3d' }}
   604→            animate={{
   605→              rotateX: isMobile ? 35 : 30 + mousePos.y * -8,
   606→              rotateY: isMobile ? 0 : mousePos.x * 8,
   607→            }}
   608→            transition={{ type: 'spring', stiffness: 50, damping: 30 }}
   609→          >
   610→            {/* Particle background */}
   611→            <ParticleCanvas />
   612→
   613→            {/* ── 3D ORBIT RINGS (SVG) ── */}
   614→            <div className="absolute inset-0" style={{ transform: 'perspective(800px) rotateX(60deg)', transformStyle: 'preserve-3d' }} aria-hidden="true">
   615→              {/* Outer ring — stats */}
   616→              <OrbitRing radius={outerRadius} color="#D4AF37" dashArray="8 6" opacity={0.25} rotateSpeed={80} strokeWidth={1.5} />
   617→              {/* Outer ring glow */}
   618→              <OrbitRing radius={outerRadius} color="#D4AF37" opacity={0.06} rotateSpeed={80} strokeWidth={8} />
   619→              {/* Inner ring — services */}
   620→              <OrbitRing radius={innerRadius} color="#5d2c86" opacity={0.2} rotateSpeed={55} reverse strokeWidth={1.5} />
   621→              {/* Inner ring glow */}
   622→              <OrbitRing radius={innerRadius} color="#5d2c86" opacity={0.05} rotateSpeed={55} reverse strokeWidth={6} />
   623→              {/* Extra decorative ring */}
   624→              <OrbitRing radius={outerRadius + 20} color="#D4AF37" dashArray="2 12" opacity={0.1} rotateSpeed={120} reverse strokeWidth={0.8} />
   625→            </div>
   626→
   627→            {/* ═══ FLAT OVERLAY — Service & Stat nodes positioned normally ═══ */}
   628→            <div className="absolute inset-0" style={{ transformStyle: 'preserve-3d' }}>
   629→              {/* ── OUTER STAT NODES ── */}
   630→              {services.map((service, i) => {
   631→                const pos = nodePositions[i];
   632→                return (
   633→                  <motion.div
   634→                    key={`stat-${i}`}
   635→                    className="absolute z-20"
   636→                    style={{
   637→                      left: `${pos.outer.x}%`,
   638→                      top: `${pos.outer.y}%`,
   639→                      transform: 'translate(-50%, -50%)',
   640→                      transformStyle: 'preserve-3d',
   641→                    }}
   642→                    initial={{ scale: 0, opacity: 0, y: 20 }}
   643→                    animate={isInView ? { scale: 1, opacity: 1, y: 0 } : {}}
   644→                    transition={{ duration: 0.5, delay: 1.0 + i * 0.1, type: 'spring' }}
   645→                  >
   646→                    <StatNode
   647→                      service={service}
   648→                      isActive={activeIndex === i}
   649→                      isHovered={hoveredIndex === i}
   650→                      onHover={() => setHoveredIndex(i)}
   651→                      onLeave={() => setHoveredIndex(null)}
   652→                      onClick={() => setActiveIndex(activeIndex === i ? null : i)}
   653→                    />
   654→                  </motion.div>
   655→                );
   656→              })}
   657→
   658→              {/* ── INNER SERVICE NODES ── */}
   659→              {services.map((service, i) => {
   660→                const pos = nodePositions[i];
   661→                return (
   662→                  <motion.div
   663→                    key={`service-${i}`}
   664→                    className="absolute z-30"
   665→                    style={{
   666→                      left: `${pos.inner.x}%`,
   667→                      top: `${pos.inner.y}%`,
   668→                      transform: 'translate(-50%, -50%)',
   669→                      transformStyle: 'preserve-3d',
   670→                    }}
   671→                    initial={{ scale: 0, opacity: 0, y: 15 }}
   672→                    animate={isInView ? { scale: 1, opacity: 1, y: 0 } : {}}
   673→                    transition={{ duration: 0.6, delay: 0.6 + i * 0.12, type: 'spring' }}
   674→                  >
   675→                    <ServiceNode
   676→                      service={service}
   677→                      isActive={activeIndex === i}
   678→                      isHovered={hoveredIndex === i}
   679→                      onHover={() => setHoveredIndex(i)}
   680→                      onLeave={() => setHoveredIndex(null)}
   681→                      onClick={() => setActiveIndex(activeIndex === i ? null : i)}
   682→                    />
   683→                  </motion.div>
   684→                );
   685→              })}
   686→
   687→              {/* ── CONNECTOR LINES ── */}
   688→              {services.map((_, i) => {
   689→                const pos = nodePositions[i];
   690→                const isActiveConn = activeIndex === i || hoveredIndex === i;
   691→                return (
   692→                  <svg
   693→                    key={`conn-${i}`}
   694→                    className="absolute inset-0 w-full h-full pointer-events-none z-10"
   695→                    viewBox="0 0 100 100"
   696→                    preserveAspectRatio="none"
   697→                  >
   698→                    <motion.line
   699→                      x1={pos.inner.x}
   700→                      y1={pos.inner.y}
   701→                      x2={pos.outer.x}
   702→                      y2={pos.outer.y}
   703→                      stroke={isActiveConn ? '#D4AF37' : '#5d2c86'}
   704→                      strokeWidth={isActiveConn ? 0.3 : 0.15}
   705→                      strokeDasharray="1 2"
   706→                      opacity={isActiveConn ? 0.6 : 0.15}
   707→                      initial={{ pathLength: 0 }}
   708→                      animate={isInView ? { pathLength: 1 } : {}}
   709→                      transition={{ duration: 0.8, delay: 1.5 + i * 0.1 }}
   710→                    />
   711→                  </svg>
   712→                );
   713→              })}
   714→
   715→              {/* ── CENTER CORE ── */}
   716→              <CenterCore isInView={isInView} />
   717→
   718→              {/* ── RING LABELS (positioned on the flat overlay) ── */}
   719→              <motion.div
   720→                className="absolute z-10"
   721→                style={{ left: '50%', top: isMobile ? '5%' : '4%', transform: 'translate(-50%, -50%)' }}
   722→                initial={{ opacity: 0, y: 10 }}
   723→                animate={isInView ? { opacity: 1, y: 0 } : {}}
   724→                transition={{ duration: 0.6, delay: 1.8 }}
   725→              >
   726→                <span className="text-[8px] sm:text-[9px] tracking-[0.25em] font-sans-body font-bold text-[#D4AF37]/60 whitespace-nowrap">
   727→                  STATISTICS & METRICS
   728→                </span>
   729→              </motion.div>
   730→
   731→              <motion.div
   732→                className="absolute z-10"
   733→                style={{ left: '50%', bottom: isMobile ? '14%' : '12%', transform: 'translate(-50%, 50%)' }}
   734→                initial={{ opacity: 0, y: -10 }}
   735→                animate={isInView ? { opacity: 1, y: 0 } : {}}
   736→                transition={{ duration: 0.6, delay: 2.0 }}
   737→              >
   738→                <span className="text-[8px] sm:text-[9px] tracking-[0.25em] font-sans-body font-bold text-[#5d2c86]/50 whitespace-nowrap">
   739→                  SERVICE CATEGORIES
   740→                </span>
   741→              </motion.div>
   742→            </div>
   743→          </motion.div>
   744→        </div>
   745→      </div>
   746→
   747→      {/* ── MOBILE CARD LAYOUT ── */}
   748→      <div className="sm:hidden mt-8 space-y-3">
   749→        {services.map((service, i) => {
   750→          const Icon = service.icon;
   751→          const isActive = activeIndex === i;
   752→          return (
   753→            <motion.button
   754→              key={`mobile-${i}`}
   755→              className={`w-full text-left rounded-xl p-4 border transition-all duration-300 ${
   756→                isActive
   757→                  ? 'bg-[#5d2c86]/5 border-[#5d2c86]/20 shadow-md'
   758→                  : 'bg-white/80 border-gray-100'
   759→              }`}
   760→              onClick={() => setActiveIndex(isActive ? null : i)}
   761→              initial={{ opacity: 0, y: 10 }}
   762→              animate={{ opacity: 1, y: 0 }}
   763→              transition={{ delay: i * 0.08 }}
   764→            >
   765→              <div className="flex items-center gap-3">
   766→                <div
   767→                  className={`w-10 h-10 rounded-full flex items-center justify-center shrink-0 ${
   768→                    isActive ? 'bg-[#5d2c86]' : 'bg-[#5d2c86]/10'
   769→                  }`}
   770→                >
   771→                  <Icon className={`w-5 h-5 ${isActive ? 'text-white' : 'text-[#5d2c86]'}`} />
   772→                </div>
   773→                <div className="flex-1 min-w-0">
   774→                  <div className="flex items-center justify-between gap-2">
   775→                    <h4 className="text-sm font-bold text-[#1A1A1A] font-sans-body">{service.title}</h4>
   776→                    <span className="text-xs font-bold font-serif-display text-[#D4AF37] whitespace-nowrap">
   777→                      {service.stat} {service.statLabel}
   778→                    </span>
   779→                  </div>
   780→                  {isActive && (
   781→                    <motion.p
   782→                      initial={{ height: 0, opacity: 0 }}
   783→                      animate={{ height: 'auto', opacity: 1 }}
   784→                      className="text-xs text-gray-500 leading-relaxed font-sans-body mt-2"
   785→                    >
   786→                      {service.description}
   787→                    </motion.p>
   788→                  )}
   789→                </div>
   790→              </div>
   791→            </motion.button>
   792→          );
   793→        })}
   794→      </div>
   795→
   796→      {/* ── DESKTOP DETAIL PANEL ── */}
   797→      <AnimatePresence>
   798→        {selectedService && (
   799→          <motion.div
   800→            initial={{ opacity: 0, y: 25, height: 0 }}
   801→            animate={{ opacity: 1, y: 0, height: 'auto' }}
   802→            exit={{ opacity: 0, y: 15, height: 0 }}
   803→            transition={{ duration: 0.4, ease: 'easeOut' }}
   804→            className="mt-8 hidden sm:block"
   805→          >
   806→            <div className="relative bg-white/95 dark:bg-[#1E1E1E]/95 backdrop-blur-md rounded-2xl shadow-xl border border-[#5d2c86]/10 overflow-hidden">
   807→              {/* Top accent bar */}
   808→              <div className="h-1 bg-gradient-to-r from-[#5d2c86] via-[#D4AF37] to-[#5d2c86]" />
   809→
   810→              <div className="p-6 flex flex-col sm:flex-row gap-6 items-start">
   811→                {/* Icon */}
   812→                <div className="w-14 h-14 rounded-2xl bg-gradient-to-br from-[#5d2c86] to-[#7d44a8] flex items-center justify-center shrink-0 shadow-lg">
   813→                  <selectedService.icon className="w-7 h-7 text-white" />
   814→                </div>
   815→
   816→                {/* Text */}
   817→                <div className="flex-1">
   818→                  <h3 className="text-xl font-bold text-[#1A1A1A] dark:text-white font-sans-body mb-2">
   819→                    {selectedService.title}
   820→                  </h3>
   821→                  <p className="text-sm text-gray-500 dark:text-gray-400 leading-relaxed font-sans-body mb-4">
   822→                    {selectedService.description}
   823→                  </p>
   824→
   825→                  {/* Stats row */}
   826→                  <div className="flex items-center gap-6">
   827→                    <div>
   828→                      <span className="text-2xl font-bold font-serif-display text-[#5d2c86] dark:text-[#7d44a8]">
   829→                        {selectedService.stat}
   830→                      </span>
   831→                      <span className="text-xs text-gray-400 dark:text-gray-500 font-sans-body ml-2 uppercase tracking-wider">
   832→                        {selectedService.statLabel}
   833→                      </span>
   834→                    </div>
   835→                    <button
   836→                      onClick={() => setActiveIndex(null)}
   837→                      className="text-xs text-[#5d2c86] hover:text-[#D4AF37] font-sans-body font-medium transition-colors"
   838→                    >
   839→                      ← Back to overview
   840→                    </button>
   841→                  </div>
   842→                </div>
   843→              </div>
   844→            </div>
   845→          </motion.div>
   846→        )}
   847→      </AnimatePresence>
   848→    </div>
   849→  );
   850→}
