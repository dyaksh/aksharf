   304в†’   304в†’function AnimatedCounter({ target, suffix = '', duration = 2 }: { target: number; suffix?: string; duration?: number }) {
   305в†’   305в†’  const [count, setCount] = useState(0);
   306в†’   306в†’  const ref = useRef<HTMLSpanElement>(null);
   307в†’   307в†’  const isInView = useInView(ref, { once: true, margin: '-40px' });
   308в†’   308в†’
   309в†’   309в†’  useEffect(() => {
   310в†’   310в†’    if (!isInView) return;
   311в†’   311в†’    let start = 0;
   312в†’   312в†’    const startTime = performance.now();
   313в†’   313в†’    const step = (now: number) => {
   314в†’   314в†’      const elapsed = now - startTime;
   315в†’   315в†’      const progress = Math.min(elapsed / (duration * 1000), 1);
   316в†’   316в†’      // Ease-out cubic
   317в†’   317в†’      const eased = 1 - Math.pow(1 - progress, 3);
   318в†’   318в†’      const current = Math.round(eased * target);
   319в†’   319в†’      if (current !== start) {
   320в†’   320в†’        setCount(current);
   321в†’   321в†’        start = current;
   322в†’   322в†’      }
   323в†’   323в†’      if (progress < 1) {
   324в†’   324в†’        requestAnimationFrame(step);
   325в†’   325в†’      }
   326в†’   326в†’    };
   327в†’   327в†’    requestAnimationFrame(step);
   328в†’   328в†’  }, [isInView, target, duration]);
   329в†’   329в†’
   330в†’   330в†’  return <span ref={ref}>{count}{suffix}</span>;
   331в†’   331в†’}
